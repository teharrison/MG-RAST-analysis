print '[ '+', '.join(groups)+' ]'​
colors = google_palette(len(groups))
group_colors = dict([(groups[i], colors[i]) for i in range(len(groups))])
print '[ '+', '.join(groups)+' ]'
 
def color_groups(args):
    for s in args['data']['series']:
        mg = s['name']
        s['name'] = grouped_mgs[mg]
        s['color'] = group_colors[grouped_mgs[mg]]
    return args
